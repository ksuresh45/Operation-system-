#include <stdio.h>
#include <fcntl.h>    // For open()
#include <unistd.h>   // For read(), write(), close()
#include <stdlib.h>   // For exit()

#define BUFFER_SIZE 1024

int main() {
    int src_fd, dest_fd;
    char buffer[BUFFER_SIZE];
    ssize_t bytesRead;

    // Open source file for reading
    src_fd = open("source.txt", O_RDONLY);
    if (src_fd < 0) {
        perror("Failed to open source file");
        exit(1);
    }

    // Open destination file for writing (create if it doesn't exist)
    dest_fd = open("destination.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (dest_fd < 0) {
        perror("Failed to open destination file");
        close(src_fd);
        exit(1);
    }

    // Read from source and write to destination
    while ((bytesRead = read(src_fd, buffer, BUFFER_SIZE)) > 0) {
        if (write(dest_fd, buffer, bytesRead) != bytesRead) {
            perror("Write error");
            close(src_fd);
            close(dest_fd);
            exit(1);
        }
    }

    if (bytesRead < 0) {
        perror("Read error");
    }

    // Close files
    close(src_fd);
    close(dest_fd);

    printf("File copied successfully.\n");
    return 0;
}
