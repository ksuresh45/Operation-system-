#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    int choice;
    char filename[50];
    FILE *fp;
    char data[100];
    
    while (1) {
        printf("\nFile Management Operations:\n");
        printf("1. Create and Write File\n");
        printf("2. Read File\n");
        printf("3. Append to File\n");
        printf("4. Delete File\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);
        getchar(); // Consume newline

        switch(choice) {
            case 1:
                printf("Enter file name to create: ");
                scanf("%s", filename);
                getchar();
                fp = fopen(filename, "w"); // Create or overwrite
                if (!fp) {
                    perror("Error creating file");
                    break;
                }
                printf("Enter data to write to file: ");
                fgets(data, sizeof(data), stdin);
                fputs(data, fp);
                fclose(fp);
                printf("File '%s' created and data written successfully.\n", filename);
                break;

            case 2:
                printf("Enter file name to read: ");
                scanf("%s", filename);
                fp = fopen(filename, "r");
                if (!fp) {
                    perror("Error opening file");
                    break;
                }
                printf("File content:\n");
                while (fgets(data, sizeof(data), fp) != NULL)
                    printf("%s", data);
                fclose(fp);
                break;

            case 3:
                printf("Enter file name to append: ");
                scanf("%s", filename);
                getchar();
                fp = fopen(filename, "a"); // Open for appending
                if (!fp) {
                    perror("Error opening file");
                    break;
                }
                printf("Enter data to append: ");
                fgets(data, sizeof(data), stdin);
                fputs(data, fp);
                fclose(fp);
                printf("Data appended successfully.\n");
                break;

            case 4:
                printf("Enter file name to delete: ");
                scanf("%s", filename);
                if (remove(filename) == 0)
                    printf("File '%s' deleted successfully.\n", filename);
                else
                    perror("Error deleting file");
                break;

            case 5:
                exit(0);

            default:
                printf("Invalid choice! Try again.\n");
        }
    }

    return 0;
}
