#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

void* print_even(void* arg) {
    for (int i = 0; i <= 10; i += 2) {
        printf("Even Thread: %d\n", i);
        sleep(1);
    }
    pthread_exit(NULL);
}

void* print_odd(void* arg) {
    for (int i = 1; i <= 10; i += 2) {
        printf("Odd Thread: %d\n", i);
        sleep(1);
    }
    pthread_exit(NULL);
}

int main() {
    pthread_t t1, t2;

    // Create threads
    if (pthread_create(&t1, NULL, print_even, NULL) != 0) {
        perror("Failed to create thread t1");
        exit(1);
    }

    if (pthread_create(&t2, NULL, print_odd, NULL) != 0) {
        perror("Failed to create thread t2");
        exit(1);
    }

    // Wait for threads to finish
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    printf("Main: Both threads finished execution.\n");
    return 0;
}
