#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_FILES 100
#define NAME_LENGTH 50

char directory[MAX_FILES][NAME_LENGTH]; // Array to store file names
int fileCount = 0;

// Function to create a file
void createFile() {
    if (fileCount >= MAX_FILES) {
        printf("Directory full! Cannot create more files.\n");
        return;
    }

    char name[NAME_LENGTH];
    printf("Enter file name to create: ");
    scanf("%s", name);

    // Check if file already exists
    for (int i = 0; i < fileCount; i++) {
        if (strcmp(directory[i], name) == 0) {
            printf("File already exists!\n");
            return;
        }
    }

    strcpy(directory[fileCount], name);
    fileCount++;
    printf("File '%s' created successfully.\n", name);
}

// Function to delete a file
void deleteFile() {
    if (fileCount == 0) {
        printf("Directory is empty!\n");
        return;
    }

    char name[NAME_LENGTH];
    printf("Enter file name to delete: ");
    scanf("%s", name);

    int found = 0;
    for (int i = 0; i < fileCount; i++) {
        if (strcmp(directory[i], name) == 0) {
            found = 1;
            // Shift remaining files
            for (int j = i; j < fileCount - 1; j++)
                strcpy(directory[j], directory[j + 1]);
            fileCount--;
            printf("File '%s' deleted successfully.\n", name);
            break;
        }
    }

    if (!found)
        printf("File '%s' not found!\n", name);
}

// Function to display all files
void displayFiles() {
    if (fileCount == 0) {
        printf("Directory is empty!\n");
        return;
    }

    printf("Files in directory:\n");
    for (int i = 0; i < fileCount; i++)
        printf("%d. %s\n", i + 1, directory[i]);
}

// Function to search for a file
void searchFile() {
    if (fileCount == 0) {
        printf("Directory is empty!\n");
        return;
    }

    char name[NAME_LENGTH];
    printf("Enter file name to search: ");
    scanf("%s", name);

    for (int i = 0; i < fileCount; i++) {
        if (strcmp(directory[i], name) == 0) {
            printf("File '%s' found in directory.\n", name);
            return;
        }
    }
    printf("File '%s' not found in directory.\n", name);
}

int main() {
    int choice;

    while (1) {
        printf("\nSingle-Level Directory Operations:\n");
        printf("1. Create File\n");
        printf("2. Delete File\n");
        printf("3. Display Files\n");
        printf("4. Search File\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1: createFile(); break;
            case 2: deleteFile(); break;
            case 3: displayFiles(); break;
            case 4: searchFile(); break;
            case 5: exit(0);
            default: printf("Invalid choice! Try again.\n");
        }
    }

    return 0;
}
